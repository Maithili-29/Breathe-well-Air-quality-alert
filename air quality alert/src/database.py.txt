import sqlite3

conn = sqlite3.connect("aqi_data.db")
c = conn.cursor()
c.execute("""CREATE TABLE IF NOT EXISTS aqi_history (
                id INTEGER PRIMARY KEY,
                location TEXT,
                aqi INTEGER,
                timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
            )""")
conn.commit()

def save_data(location, aqi):
    c.execute("INSERT INTO aqi_history(location, aqi) VALUES (?, ?)", (location, aqi))
    conn.commit()

def fetch_history():
    c.execute("SELECT * FROM aqi_history ORDER BY timestamp DESC")
    return c.fetchall()
